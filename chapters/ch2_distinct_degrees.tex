\section{Subgraphs with Many Degrees}
\label{sec:ch2}

In this chapter, we present the following result of Bukh and Sudakov.

\begin{theorem}[Bukh--Sudakov {\cite{BUKH2007612}}]
  \label{thm:bs2007}
  Every Ramsey graph on $n$ vertices induces a subgraph on $\Omega(n)$ vertices with $\Omega(\sqrt{n})$ distinct degrees.
\end{theorem}

This was originally conjectured by Erdős, Faudree, and Sós \cite{EFS} in light of the observation that the random graph $G_{n, 1/2}$ exhibits such a property almost surely, but the $\Omega(\sqrt{n})$ bound is in fact not optimal for both Ramsey graphs and $G_{n, 1/2}$. In particular, an upper bound of $O(n^{2/3})$ is noted in \cite{BUKH2007612}, which was later proved to be the correct order for Ramsey graphs by Jenssen, Keevash, Long, and Yepremyan \cite{jenssen2019distinctdegreesinducedsubgraphs} via a continuous relaxation approach. The corresponding order for $G_{n, 1/2}$ was also matched in an unpublished work of Conlon, Morris, Samotij, and Saxton \cite{conlon2019distinctdegrees}, further reinforcing the case that Ramsey graphs resemble random graphs in various respects. We will, however, focus on the original result of Bukh and Sudakov, as it introduces a key characteristic of Ramsey graphs that will be insightful in the following chapter.

\begin{definition}
  \label{def:diversity}

  An $n$-vertex graph $G$ is \textit{$(\eps, \delta)$-diverse} if for $x \in V(G)$, we have $|N(x) \Delta N(y)| \geq \eps n$ for all but at most $n^{\delta}$ vertices $y \in V(G)$.
\end{definition}

The notion of diversity captures the idea that the neighbourhoods of most pairs of vertices differ significantly. In particular, $(\eps, \delta)$-diversity directly conditions our assumption that $|N(x) \Delta N(y)| = D \geq \eps n$ in the discussion of the anticoncentration principle in Section \ref{sec:anticoncentration}. Hence, following our discussion in Section \ref{sec:anticoncentration}, we see that $(\eps, \delta)$-diversity implies that $\prob[\deg_U(x) = \deg_U(y)] = O(1/\sqrt{n})$ for all but at most $n^{\delta}$ pairs of vertices $x, y \in V(G)$, from which we use Jensen's inequality to conclude that there are many distinct degrees in a diverse graph. The following lemma formalizes this observation and reduces Theorem \ref{thm:bs2007} to showing every Ramsey graph contains a diverse subgraph of linear size.

\begin{lemma}
  \label{lemma:diversity-degrees}

  Let $\delta \in (0, 1/2)$ and let $G$ be an $(\eps, \delta)$-diverse graph on $n$ vertices for large enough $n$. Then for $n/4 \leq m \leq 3n/4$, there is a subgraph on $m$ vertices containing vertices of $\sqrt{\eps n}/320$ distinct degrees.
\end{lemma}

\begin{proof}
  Fix $m$ as above. Let $U$ be chosen uniformly at random among all subsets of $V(G)$ of size $m$. Let $H = G[U]$. It suffices to show that with positive probability, $H$ contains $\Omega(\sqrt{n})$ vertices of distinct degrees.

  For distinct $x, y \in V(G)$, define $\mathbbm{1}_{xy}$ to be the indicator for the event that $x, y \in V(H)$ and $\deg_H(x) = \deg_H(y)$. Then $\E[\mathbbm{1}_{xy}] \leq \prob[\deg_U(x) = \deg_U(y)]$. Put $D = N_G(x) \Delta N_G(y)$, and $k = |D \cap U|$. Write $A = N_G(x) \setminus N_G(y)$ and $B = N_G(y) \setminus N_G(x)$. Then 
  \[
    \prob[\deg_U(x) = \deg_U(y)] = \prob\left[|A \cap U| = |B \cap U|\right].
  \]
  Suppose $|D| \geq \eps n$. Since $\frac{|D|}{4} \leq \frac{m|D|}{n} \leq \frac{3|D|}{4}$, Proposition \ref{prop:hypergeometric} yields
  \[
    \prob\left[k < \frac{|D|}{8} \text{ or } k > \frac{7|D|}{8}\right] \leq \prob\left[\left|k - \E[k]\right| > \frac{|D|}{8}\right] \leq e^{-\Theta(|D|)} = e^{-\Theta(n)} = o(n^{-1/2}).  
  \]
  On the other hand, recall that $U$ is a uniformly random subset of a fixed size $m$. Since every $k$-subset of $D$ is equally likely to be $U \cap D$ by symmetry, conditioning on $|U \cap D| = k$, the $k$ elements of $U$ that land in $D$ form a uniformly random subset of $D$ of size $k$. It follows that when conditioning on $k$, the event $|A \cap U| = |B \cap U| = k/2$ follows the hypergeometric distribution $\mathrm{Hyp}(|D|, |A|, k)$. Thus,
  \[
    \prob\left[\deg_U(x) = \deg_U(y) \, \middle| \, \frac{|D|}{8} \leq k \leq \frac{7|D|}{8}\right] = \frac{\binom{|A|}{k/2}\binom{|B|}{k/2}}{\binom{|D|}{k}}.
  \]
  Since $|A| + |B| = |D|$, by the log-concavity of the binomial coefficients we have $\binom{|A|}{k/2}\binom{|B|}{k/2} \leq \binom{|D|/2}{k/2}^2$. It now follows from Stirling's formula and $k(|D| - k) \geq \frac{7}{64}|D|^2$ that the above probability does not exceed
  \begin{equation}
    \frac{\binom{|D|/2}{k/2}^2}{\binom{|D|}{k}} = (1 + o(1))\sqrt{\frac{2|D|}{\pi k(|D| - k)}} \leq \frac{3}{\sqrt{|D|}} \leq \frac{3}{\sqrt{\eps n}}. \label{eq:prob-deg-diff}
  \end{equation}
  Thus if $|N_G(x) \Delta N_G(y)| \geq \eps n$ then $\E[\mathbbm{1}_{xy}] \leq 3/\sqrt{\eps n} + o(n^{-1/2}) \leq 4/\sqrt{\eps n}$.

  Let $I = \sum_{x, y \in V(G)} \mathbbm{1}_{xy}$. By definition of an $(\eps, \delta)$-diverse graph, we have $\E[\mathbbm{1}_{xy}] \leq 4/\sqrt{\eps n}$ for all but at most $n^{1 + \delta} < n^{3/2}$ pairs $x, y \in V(G)$. Thus,
  \[
    \E[I] \leq n^{1 + \delta} + \frac{4}{\sqrt{\eps n}} \cdot n^2 \leq \frac{5}{\sqrt{\eps}}n^{3/2},
  \]
  and so there is a choice for $U$ such that $I \leq 5n^{3/2}/\sqrt{\eps}$. Now let $r$ be the number of vertices of distinct degrees in $H$. Let $n_k$ be the number of vertices in $H$ with degree $k$ and note that $\sum_{k} n_k = m$. By Jensen's inequality
  \[
    5n^{3/2}/\sqrt{\eps} \geq I = \sum_{k} \binom{n_k}{2} \geq r\binom{\frac{1}{r}\sum_{k} n_k}{2} \geq \frac{m^2}{4r} \geq \frac{n^2}{64r}.
  \]
  Rearranging yields the lemma.
\end{proof}

\textit{Remark.} The hypergeometric distribution $\mathrm{Hyp}(|D|, |A|, k)$ has variance 
\[
  \sigma^2 = \frac{k|A||B|(|D| - k)}{|D|^2(D - 1)} \leq \frac{k(|D| - k)}{4(|D| - 1)},
\]
where inequality follows from AM-GM. When $k$ is not the tail end of the distribution, we have $\sigma = \Theta(\sqrt{D}) = \Theta(\sqrt{n})$, and the Stirling bound in \eqref{eq:prob-deg-diff} is asymptotically $1/(\sigma\sqrt{2\pi})$ as $n \to \infty$, which is exactly the maximum density of a normal distribution with standard deviation $\sigma$. This confirms the intuition from Section~\ref{sec:anticoncentration} that the degree difference is approximately normally distributed so its point probabilities are $O(1/\sigma) = O(1/\sqrt{n})$.

As mentioned above, it remains to show that every Ramsey graph contains a diverse subgraph of linear size. 

\begin{lemma}
  \label{lemma:diverse-subgraph}

  For $\delta \in (0, 1)$, there exists $\eps = \eps(C, \delta) > 0$ such that every $C$-Ramsey graph $G$ on $n$ vertices contains an $(\eps, \delta)$-diverse subgraph on $\Omega(n)$ vertices.
\end{lemma}

\begin{proof}
  Suppose not. Let $K = K(C)$ be a large constant to be determined. Let $c = 8^{-K}$ and $\eps = 1/K$. We inductively construct a sequence of disjoint vertex sets $S_1, S_2, \ldots, S_K$ of size $(cn)^\delta$, such that for all $S_i$ one of the following conditions holds:
  \begin{enumerate}[]
    \item[(i)] $d(S_i, S_j) < 8\eps$ for $j > i$,
    \item[(ii)] $d(S_i, S_j) > 1 - 8\eps$ for $j > i$.
  \end{enumerate}
  Alongside, we also construct a sequence of induced subgraphs $G = G_0 \supset G_1 \supset \cdots \supset G_K$, such that $|G_i| \geq |G_{i - 1}|/8$ and $S_i \subseteq V(G_{i - 1})$ for all $i$.

  We first show that this construction suffices. By working with $\overline{G}$ instead of $G$ if necessary, we may assume that $r \geq K/2$ of the sets $S_{1}, \ldots, S_{K}$ satisfy condition (i), say $S_{i_1}, \ldots, S_{i_r}$. Put $S = \bigsqcup_{k = 1}^r S_{i_k}$. Since each $S_{i_k}$ has size $(cn)^{\delta}$ and $\eps = 1/K$,
  \[
    d(S) \leq \frac{\binom{(cn)^{\delta}}{2}}{\binom{r(cn)^{\delta}}{2}} + \max_{(k, l)} d(S_{i_k}, S_{i_l}) \leq \frac{1}{r} + 8\eps \leq \frac{10}{K}.
  \]
  Thus Theorem \ref{thm:ramsey-density} yields a homogeneous set $A$ in $S$ of size
  \[
    |A| \geq \frac{\gamma K}{10\log(K/10)} \log |S| = \frac{\gamma \delta K}{10\log(K/10)} \log cn = \frac{\gamma \delta K}{10\log(K/10)} \log n + o(\log n),
  \]
  for some absolute constant $\gamma > 0$ given by the theorem. But then $\frac{\gamma \delta K}{10\log(K/10)} > C$ for sufficiently large $K$, and so $|A| > C\log n$, contradiction.
  
  We now construct the sets $S_i$ and $G_i$ inductively. Suppose $G_1, \ldots, G_{i - 1}$ and $S_1, \ldots, S_{i - 1}$ have been constructed. By induction, $|G_{i - 1}| \geq 8^{1 - i}|G| \geq cn$. Note that $G_{i - 1}$ is not $(\eps, \delta)$-diverse by assumption, so there exists a vertex $x \in V(G_{i - 1})$ and a set $S_i \subseteq G_{i - 1}$ of size $(cn)^{\delta}$ such that $|N_{G_{i - 1}}(x) \Delta N_{G_{i - 1}}(y)| < \eps|G_{i - 1}|$ for $y \in S_i$. Let $T_i = V(G_{i - 1}) \setminus S_i$. Since $(cn)^\delta \leq cn/2$ for large enough $n$, we have $|T_i| \geq |G_{i - 1}|/2$. 
  
  Suppose $|\overline{N}_{T_i}(x)| \geq |T_i|/2$. Then for $y \in S_i$,
  \[
    d(y, \overline{N}_{T_i}(x)) = \frac{|N_{T_i}(y) \cap \overline{N}_{T_i}(x)|}{|N_{T_i}(x)|} < \frac{\eps|G_{i - 1}|}{|T_i|/2} \leq 4\eps.
  \]
  Partition $\overline{N}_{T_i}(x)$ into $V_i \sqcup Z_i$, where $V_i = \{v \in \overline{N}_{T_i}(x) : d(v, S_i) < 8\eps\}$. Notice
  \[
    8\eps|Z_i| \leq \sum_{v \in Z_i} d(v, S_i) \leq \frac{e(\overline{N}_{T_i}(x), S_i)}{|S_i|} = \frac{|\overline{N}_{T_i}(x)|}{|S_i|}\sum_{y \in S_i} d(y, \overline{N}_{T_i}(x)) \leq 4\eps|\overline{N}_{T_i}(x)|,
  \]
  so $|W_i| \leq |\overline{N}_{T_i}(x)|/2$. But then $|V_i| \geq |\overline{N}_{T_i}(x)|/2 \geq |T_i|/4 \geq |G_{i - 1}|/8$. Set $G_i = G[V_i]$. Since $d(S_i, U) < 8\eps$ for all $U \subseteq V_i$, condition (i) holds for $S_i$. In the case that $|N_{T_i}(x)| \geq |T_i|/2$, the same argument applies by setting $V_i = \{v \in N_{T_i}(x) : d(v, S_i) > 1 - 8\eps\}$, which yields $|G_i| \geq |G_{i - 1}|/8$ and $S_i$ satisfying condition (ii). This completes the induction.
\end{proof}

Theorem \ref{thm:bs2007} follows immediately from Lemmas \ref{lemma:diversity-degrees} and \ref{lemma:diverse-subgraph}.
